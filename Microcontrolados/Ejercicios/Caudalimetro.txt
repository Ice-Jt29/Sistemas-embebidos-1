#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main() {
    char linea[100];
    int ultimo, mayor = -1, menor = 100, suma = 0, contador = 0;

    while (fgets(linea, sizeof(linea), stdin)) {
        // Buscar formato {'caudal': VALOR}
        int valor;
        // Intentar extraer el valor con sscanf
        if (sscanf(linea, "{'caudal': %d}", &valor) == 1) {
            // Validar rango 0-99
            if (valor >= 0 && valor <= 99) {
                ultimo = valor;
                suma += valor;
                contador++;
                if (valor > mayor) mayor = valor;
                if (valor < menor) menor = valor;

                float promedio = (float)suma / contador;
                printf("{'último': %d, 'mayor': %d, 'menor': %d, 'promedio': %.2f}\n",
                       ultimo, mayor, menor, promedio);
            }
            // Si no está en rango, ignorar
        }
        // Si no cumple formato, ignorar
    }
    return 0;
}
