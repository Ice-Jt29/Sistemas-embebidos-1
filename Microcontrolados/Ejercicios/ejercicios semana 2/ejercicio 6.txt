#include <stdio.h>

int main(void) {
    int NUMERO, CANTIDAD;

    // Leer número de integrantes y cantidad de exponentes
    if (scanf("%d", &NUMERO) != 1) return 0;
    if (scanf("%d", &CANTIDAD) != 1) return 0;

    // Validar que la cantidad sea positiva y razonable
    if (CANTIDAD <= 0 || CANTIDAD > 100) return 0; // Puedes ajustar el límite

    int ORDEN[100]; // Arreglo para los exponentes

    // Leer los exponentes
    for (int i = 0; i < CANTIDAD; i++) {
        if (scanf("%d", &ORDEN[i]) != 1) return 0;
    }

    // Imprimir resultado en el formato pedido
    printf("{'Número de integrantes en el grupo': %d, ", NUMERO);
    printf("'Cantidad de exponentes': %d, ", CANTIDAD);
    printf("'Orden de exponentes': [");
    for (int i = 0; i < CANTIDAD; i++) {
        printf("%d", ORDEN[i]);
        if (i < CANTIDAD - 1) printf(", ");
    }
    printf("]}\n");

    return 0;
}
