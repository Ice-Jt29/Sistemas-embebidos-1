Descripción:
Se puede calcular el cuadrado de un número N al sumar los primeros N números impares.
Con esta lógica, el cuadrado del número 7 es la sumatoria de los primeros 7 números impares, es decir 1+3+7+8+9+11+13=49
Describir una solución para el microcontrolador que calcule el cuadrado de un número recibido por puerto serial e imprima el resultado por puerto serial.
Si se recibe algo diferente a un número entero positivo mayor a cero se debe ignorar.*/




#include <stdio.h>
#include <ctype.h>

// Función para validar si una cadena representa un entero positivo
int es_entero_positivo(const char* s) {
    if (*s == '\0') return 0;
    while (*s) {
        if (!isdigit(*s)) return 0;
        s++;
    }
    return 1;
}

int main() {
    char entrada[100];
    int N;

    printf("Ingrese un número entero positivo: ");
    if (fgets(entrada, sizeof(entrada), stdin)) {
        // Eliminar salto de línea
        char *p = entrada;
        while (*p && *p != '\n') p++;
        *p = '\0';

        if (es_entero_positivo(entrada)) {
            N = atoi(entrada);
            if (N > 0) {
                int suma = 0;
                int impar = 1;
                for (int i = 0; i < N; i++) {
                    suma += impar;
                    impar += 2;
                }
                printf("El cuadrado de %d es %d\n", N, suma);
            }
        }
        // Si no es válido, no se imprime nada (se ignora)
    }
    return 0;
}
